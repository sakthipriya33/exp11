Open repo on GitHub
Go to Settings → Secrets and variables → Actions
Click New repository secret
Add:

Name: API_KEY  
Value: 12345


name: Secrets Example

on:
  push:

jobs:
  demo:
    runs-on: ubuntu-latest

    steps:
      - name: Use Secrets
        env:
          API_KEY: ${{ secrets.API_KEY }}
          DB_PASSWORD: ${{ secrets.DB_PASSWORD }}
        run: echo "Secrets used securely"
